# mac

  1. put localxx serverxx //sftp 文件传输
  2. ssh -p 端口号 username@服务器ip //ssh登录服务器
  3. scp -P端口号 本地文件路径 username@服务器ip:目的路径 //上传文件至服务器
  4. scp -P端口号 username@ip:路径 本地路径 //从服务器下载文件到本地

# mysql

  1. mysql -h localhost -u root -p //连接数据库
  2. show databases; //显示数据库
  3. use databasename; //使用databasename数据库
  4. show tables; //显示数据表
  5. describe tablename; //显示tablename表结构
  6. create database name; //创建数据库
  7. drop database name //直接删除数据库，不提醒
  8. if exists() end if; //判断是够存在,（）中可以写select语句
  9. select a into @a from table where =; //从表中某条数据中取出某字段
  10. select count(*) into @allcnt from table; //取出表中数据的个数
  11. call function(a,b,c); //执行函数
  12. select date_format(NOW(),'TR%Y%m%d%H%i%s'); //日期转字符串，注意赋值的时侯（）包起来
  13. set a=now(); //赋值语句
  14. select GROUP_CONCAT(d.depotid separator ',') waterstation from t_water_user_bind d
  15. SELECT LAST_INSERT_ID(); //插入的自增数据值
  16. union all //合并两个表
  17. group by name //去重
  18. distinct name //去重
  19. to_days

# pm2

  1. pm2 start app.js -i 4 // 后台运行pm2，启动4个app.js
  2. pm2 start app.js --name my-api // 命名进程
  3. pm2 list // 显示所有进程状态
  4. pm2 monit // 监视所有进程
  5. pm2 logs // 显示所有进程日志
  6. pm2 stop all // 停止所有进程
  7. pm2 restart all // 重启所有进程
  8. pm2 reload all // 0秒停机重载进程 (用于 NETWORKED 进程)
  9. pm2 stop 0 // 停止指定的进程
  10. pm2 restart 0 // 重启指定的进程
  11. pm2 startup // 产生 init 脚本 保持进程活着
  12. pm2 web // 运行健壮的 computer API endpoint
  13. pm2 delete 0 // 杀死指定的进程
  14. pm2 delete all // 杀死全部进程

# docker

  1. docker run -d ubuntu /bin/sh -c "while true; do echo hello world; sleep 1; done" //创建一个以进程方式运行的容器
  2. docker ps // 查看所有容器
  3. docker logs 容器id/名称 // 查看容器日志
  4. docker stop 容器id/名称 // stop容器
  5. docker start/restart 容器id/名称 // start/restart容器
  6. docker rm 容器id/名称 // 删除容器，需在停止状态下
  7. docker rmi 镜像id
  8. docker port 容器id/名称 // 查看容器端口
  9. docker top 容器id/名称 // 查看容器内部运行的进程
  10. docker inspect 容器id/名称 // 查看Docker的底层信息
  11. docker images // 列出本地主机上的镜像
  12. docker pull XX // 下载镜像
  13. docker search XX // 搜索镜像
  14. docker run XX // 使用镜像
  15. touch Dockerfile // 新建Dockerfile文件
  16. docker build -t tomcatname // 根据Dockerfile创建镜像
  17. docker run -p 8080:8080 tomcatname // 运行镜像
  18. docker exec -it mysql bash
  19. docker run -i -t --name HelloDocker -p 80:8080 tomcat /bin/bash
  20. docker inspect -f '**.Id' HelloDocker // 获取id全称
  21. docker cp 本地文件路径 ID全称:容器路径 // 本机和容器之间的文件传输
  22. docker cp smopark.war 38efcb8cb22ff0433667a0a593dab0348453d28dc2a04a72d794f9987540cc45:/usr/local/tomcat/webapps
